        var ctx = canvas.getContext('2d');
        canvas.height = 600;
        canvas.width = 500;
        var pos = {
            x: 0
            , y: 50
        };

        function init() {
            draw();
        }

        function draw() {
            pos.x = pos.x + 5;
            if (pos.x > canvas.width) {
                pos.x = 0;
            }
            if (pos.y > canvas.height) {
                pos.y = 0;
            }
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.beginPath();
            ctx.fillStyle = "red";
            ctx.arc(pos.x, pos.y, 50, 0, Math.PI * 2);
            ctx.fill();
            ctx.closePath();
            window.setTimeout(draw, 50);
        }
    </script>
